Procedimento fettuccineSequence(month, day, length: inteiro)
Var
    i: inteiro
Inicio
    sequence[1] <- month
    sequence[2] <- day
    para i de 3 ate length faca
        se (i % 2 = 0) entao
            sequence[i] <- sequence[i-2] - sequence[i-1]
        senao
            sequence[i] <- sequence[i-2] + sequence[i-1]
        fimse
    fimpara
FimProcedimento

Procedimento primeNumbersFromAge(age, length: inteiro)
Var
    num, count, i: inteiro
    ePrimo: logico
Inicio
    se (age > 1) entao
        num <- age
    senao
        num <- 2
    fimse

    count <- 1

    enquanto (count <= length) faca
        ePrimo <- verdadeiro
        para i de 2 ate (num - 1) faca
            se (num % i = 0) entao
                ePrimo <- falso
            fimse
        fimpara
        se (ePrimo) entao
            sequence[count] <- num
            count <- count + 1
        fimse
        num <- num + 1
    fimenquanto
FimProcedimento

    start, count: inteiro
