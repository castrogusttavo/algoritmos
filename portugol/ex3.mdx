Var
    matrix: vetor[1..3, 1..3] de inteiro
    option: caractere
    positions, age, n, i, j: inteiro

Procedimento shiftMatrix(positions: inteiro)
Var
    flatten, shifted: vetor[1..9] de inteiro
Inicio
    // Achatar a matriz em um vetor
    para i de 1 ate 3 faca
        para j de 1 ate 3 faca
            flatten[(i-1)*3+j] <- matrix[i,j]
        fimpara
    fimpara

    // Deslocar os elementos
    para i de 1 ate 9 faca
        shifted[(i+positions-1) mod 9 + 1] <- flatten[i]
    fimpara

    // Reconstituir a matriz
    para i de 1 ate 3 faca
        para j de 1 ate 3 faca
            matrix[i,j] <- shifted[(i-1)*3+j]
        fimpara
    fimpara
FimProcedimento

Procedimento doubleMatrix()
Inicio
    para i de 1 ate 3 faca
        para j de 1 ate 3 faca
            matrix[i,j] <- matrix[i,j] * 2
        fimpara
    fimpara
FimProcedimento

Procedimento filterMatrix(age: inteiro)
Inicio
    para i de 1 ate 3 faca
        para j de 1 ate 3 faca
            se matrix[i,j] >= age entao
                matrix[i,j] <- matrix[i,j]
            senao
                matrix[i,j] <- 0
            fimse
        fimpara
    fimpara
FimProcedimento
