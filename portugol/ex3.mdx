Procedimento shiftMatrix(posicoes: inteiro)
Var
    flattened, shifted: vetor[1..9] de inteiro
Inicio
    // Achatar a matriz em um vetor
    para i de 1 ate 3 faca
        para j de 1 ate 3 faca
            flattened[(i-1)*3+j] <- matrix[i,j]
        fimpara
    fimpara

    // Deslocar os elementos
    para i de 1 ate 9 faca
        shifted[(i+posicoes-1) mod 9 + 1] <- flattened[i]
    fimpara

    // Reconstituir a matriz
    para i de 1 ate 3 faca
        para j de 1 ate 3 faca
            matrix[i,j] <- shifted[(i-1)*3+j]
        fimpara
    fimpara
FimProcedimento
