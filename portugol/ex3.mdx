Var
    matriz: vetor[1..3, 1..3] de inteiro
    opcao: caractere
    posicoes, idade, n, i, j: inteiro

Procedimento deslocarMatriz(posicoes: inteiro)
Var
    achatar, deslocada: vetor[1..9] de inteiro
Inicio
    // Achatar a matriz em um vetor
    para i de 1 ate 3 faca
        para j de 1 ate 3 faca
            achatar[(i-1)*3+j] <- matriz[i,j]
        fimpara
    fimpara

    // Deslocar os elementos
    para i de 1 ate 9 faca
        deslocada[(i+posicoes-1) mod 9 + 1] <- achatar[i]
    fimpara

    // Reconstituir a matriz
    para i de 1 ate 3 faca
        para j de 1 ate 3 faca
            matriz[i,j] <- deslocada[(i-1)*3+j]
        fimpara
    fimpara
FimProcedimento

Procedimento dobrarMatriz()
Inicio
    para i de 1 ate 3 faca
        para j de 1 ate 3 faca
            matriz[i,j] <- matriz[i,j] * 2
        fimpara
    fimpara
FimProcedimento

Procedimento filtrarMatriz(idade: inteiro)
Inicio
    para i de 1 ate 3 faca
        para j de 1 ate 3 faca
            se matriz[i,j] >= idade entao
                matriz[i,j] <- matriz[i,j]
            senao
                matriz[i,j] <- 0
            fimse
        fimpara
    fimpara
FimProcedimento
