Var
    tamanho, i, idade: inteiro
    sequencia: vetor[1..100] de inteiro

Procedimento gerarSequenciaFibonacci(tamanho: inteiro)
Var
    i: inteiro
Inicio
    sequencia[1] <- 0
    sequencia[2] <- 1
    para i de 3 ate tamanho faca
        sequencia[i] <- sequencia[i-1] + sequencia[i-2]
    fimpara
FimProcedimento

Procedimento gerarSequenciaFettuccine(mes, dia, tamanho: inteiro)
Var
    i: inteiro
Inicio
    sequencia[1] <- mes
    sequencia[2] <- dia
    para i de 3 ate tamanho faca
        se (i % 2 = 0) entao
            sequencia[i] <- sequencia[i-2] - sequencia[i-1]
        senao
            sequencia[i] <- sequencia[i-2] + sequencia[i-1]
        fimse
    fimpara
FimProcedimento

Procedimento gerarNumerosPrimos(idade, tamanho: inteiro)
Var
    numero, contador, i: inteiro
    ehPrimo: logico
Inicio
    se (idade > 1) entao
        numero <- idade
    senao
        numero <- 2
    fimse

    contador <- 1

    enquanto (contador <= tamanho) faca
        ehPrimo <- verdadeiro
        para i de 2 ate (numero - 1) faca
            se (numero % i = 0) entao
                ehPrimo <- falso
            fimse
        fimpara
        se (ehPrimo) entao
            sequencia[contador] <- numero
            contador <- contador + 1
        fimse
        numero <- numero + 1
    fimenquanto
FimProcedimento

Procedimento gerarNumerosPares(dia, tamanho: inteiro)
Var
    inicio, contador: inteiro
Inicio
    se (dia % 2 = 0) entao
        inicio <- dia
    senao
        inicio <- dia - 1
    fimse

    contador <- 1
    enquanto (contador <= tamanho) faca
        sequencia[contador] <- inicio
        inicio <- inicio + 2
        contador <- contador + 1
    fimenquanto
FimProcedimento

Procedimento gerarNumerosImpares(mes, tamanho: inteiro)
Var
    inicio, contador: inteiro
Inicio
    se (mes % 2 = 1) entao
        inicio <- mes
    senao
        inicio <- mes + 1
    fimse

    contador <- 1
    enquanto (contador <= tamanho) faca
        sequencia[contador] <- inicio
        inicio <- inicio + 2
        contador <- contador + 1
    fimenquanto
FimProcedimento
