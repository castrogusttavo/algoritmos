    soma, i: inteiro
inicio

        temp <- num1
        num1 <- num2
        num2 <- temp
    fimse

    para i de num1 ate num2 faca
        soma <- soma + i

funcao ProdutoExclusivo(num1: inteiro; num2: inteiro): inteiro

    produto, i: inteiro
inicio

    fimse

    retorne produto
