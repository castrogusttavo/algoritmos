Var
    number1, number2, i, age: inteiro
    result: real
    numbers: vetor[1..100] de real
    option: caractere
    powerResult, inclusiveSumResult, exclusiveProductResult: real

funcao Power(base: inteiro; exponent: inteiro): real
var
    i: inteiro
    result: real
inicio
    result <- 1
    para i de 1 ate exponent faca
        result <- result * base
    fimpara
    retorne result
fimfuncao

funcao InclusiveSum(number1: inteiro; number2: inteiro): inteiro
var
    sum, i, temp: inteiro
inicio
    se number1 > number2 entao
        temp <- number1
        number1 <- number2
        number2 <- temp
    fimse
    sum <- 0
    para i de number1 ate number2 faca
        sum <- sum + i
    fimpara
    retorne sum
fimfuncao

funcao ExclusiveProduct(number1: inteiro; number2: inteiro): inteiro
var
    product, i, temp: inteiro
inicio
    se number1 > number2 entao
        temp <- number1
        number1 <- number2
        number2 <- temp
    fimse
    product <- 1
    para i de number1 + 1 ate number2 - 1 faca
        product <- product * i
    fimpara
    retorne product
fimfuncao

Procedimento loadArray(age, first, step: inteiro)
Var
    i: inteiro
Inicio
    para i de 1 ate age faca
        numbers[i] <- first + (i - 1) * step
    fimpara
    escreval("Vetor carregado com sucesso.")
    para i de 1 ate age faca
        escreva(numbers[i], " ")
    fimpara
    escreval("")
FimProcedimento
