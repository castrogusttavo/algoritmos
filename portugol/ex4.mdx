Var
    numero1, numero2, i, idade: inteiro
    resultado: real
    numeros: vetor[1..100] de real
    opcao: caractere
    resultadoPotencia, resultadoSomaInclusiva, resultadoProdutoExclusivo: real

funcao Potencia(base: inteiro; expoente: inteiro): real
var
    i: inteiro
    resultado: real
inicio
    resultado <- 1
    para i de 1 ate expoente faca
        resultado <- resultado * base
    fimpara
    retorne resultado
fimfuncao

funcao SomaInclusiva(numero1: inteiro; numero2: inteiro): inteiro
var
    soma, i, temp: inteiro
inicio
    se numero1 > numero2 entao
        temp <- numero1
        numero1 <- numero2
        numero2 <- temp
    fimse
    soma <- 0
    para i de numero1 ate numero2 faca
        soma <- soma + i
    fimpara
    retorne soma
fimfuncao

funcao ProdutoExclusivo(numero1: inteiro; numero2: inteiro): inteiro
var
    produto, i, temp: inteiro
inicio
    se numero1 > numero2 entao
        temp <- numero1
        numero1 <- numero2
        numero2 <- temp
    fimse
    produto <- 1
    para i de numero1 + 1 ate numero2 - 1 faca
        produto <- produto * i
    fimpara
    retorne produto
fimfuncao

Procedimento carregarVetor(idade, primeiro, salto: inteiro)
Var
    i: inteiro
Inicio
    para i de 1 ate idade faca
        numeros[i] <- primeiro + (i - 1) * salto
    fimpara
    escreval("Vetor carregado com sucesso.")
    para i de 1 ate idade faca
        escreva(numeros[i], " ")
    fimpara
    escreval("")
FimProcedimento
